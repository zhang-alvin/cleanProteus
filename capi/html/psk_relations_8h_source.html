563  krn = (1.0-Se)*(1.0-Se)*(1.0-pow(Se,(2.0+lambda)/lambda));
564  dkrn = (-2.0*(1.0-Se)*(1.0-pow(Se,(2.0+lambda)/lambda))
565  -((2.0+lambda)/lambda)*(1.0-Se)*(1.0-Se)*pow(Se,(2.0+lambda)/lambda-1.0))*dSe_dSw;
566 
567  psic = pd*Semoovlmo*Se;
568  dpsic = pd*(-1.0/lambda)*Semoovlmo*dSe_dSw;
569  }
